💻 Basit Bir Yapay Nöron (Python)
# Tek katmanlı nöron modeli
def noron(girisler, agirliklar, esik):
    toplam = sum(g * a for g, a in zip(girisler, agirliklar))
    return 1 if toplam >= esik else 0

sonuc = noron([1, 0.5, 0.8], [0.4, 0.6, 0.3], 0.7)
print("Çıktı:", sonuc)  # → 1 (aktif)
